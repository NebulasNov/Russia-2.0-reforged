d_russia_reworked_show_errors = {
    log = "Russia Reworked debug :"
    if = {
        limit = {
            check_variable = {
                var = global.russia_reworked_core_file_missing
                value = 1
                compare = greater_than_or_equals
            }
        }
        log = " [?global.russia_reworked_core_file_missing] core files missing."
        log = " You should NOT play this mod with missing core files."
        log = " Any bug happening while playing with core files missing is not our responsibility."
    }
    else = {
        log = " No core file missing."
    }

    if = {
        limit = {
            check_variable = {
                var = global.russia_reworked_noncore_file_missing
                value = 1
                compare = greater_than_or_equals
            }
        }
        log = " [?global.russia_reworked_noncore_file_missing] non-core files missing."
        log = " Playing with non-core files missing is fine, but bugs and unexpected things might still happen."
    }
    else = {
        log = " No non-core file missing."
    }

    log = " Note : The result of this might not be accurate."
    log = "           You might or might not have conflicts that we did not detect."
}

            log = " Beginning total annihilation of [?args^0.GetName]..."
            FROM = {
                set_temp_variable = { nuke_drop = 3 }
                if = {
                    limit = {
                        check_variable = {
                            var = args^1
                            value = 0
                            compare = greater_than
                        }
                    }
                    set_temp_variable = { nuke_drop = args^1 }
                    clamp_temp_variable = {
                        var = nuke_drop
                        min = 1
                        max = 100
                    }
                    round_temp_variable = nuke_drop
                }
                set_temp_variable = { nuke_count = 0 }

                while_loop_effect = {
                    break = break_loop
                    limit = {
                        check_variable = {
                            var = nuke_count
                            value = var:nuke_drop
                            compare = less_than
                        }
                    }

                    launch_nuke = {
                        state = var:args^0
                        use_nuke = no
                    }
                    var:args^0 = {
                        set_temp_variable = { population_loss_count = state_population_k@var:args^0 }
                        multiply_temp_variable = { population_loss_count = 0.01 }
                        if = {
                            limit = {
                                # Avoid integer overflow
                                check_variable = {
                                    var = population_loss_count
                                    value = 2140
                                    compare = greater_than_or_equals
                                }
                            }
                            set_temp_variable = { population_loss_count = 2140000 }
                        }
                        else = {
                            multiply_temp_variable = { population_loss_count = -1000 }
                        }
                        add_manpower = var:population_loss_count
                    }
                    add_to_temp_variable = { nuke_count = 1 }
                }
                log = " Done! Unpause the game."
            }
        }
        else = {
            log = " Error :"
            log = " Argument given ([?args^0]) is not a valid state ID."
            log = " Usage : russia_reworked_tsar_bomba_nuke_debug {STATE_ID} {NUMBER_OF_NUKES}"
            log = " STATE_ID : required"
            log = " NUMBER_OF_NUKES : optional, default:3, min:1, max:100"
        }
    }
}
