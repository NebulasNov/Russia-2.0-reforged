#Add daily gain
ukr_add_daily_gain_anarchist_states = {
    for_each_loop = {
        array = ukr.anarchist_states
        value = v
        index = i
        break = break_array

        if = {
            limit = {
                check_variable = {
                    var = ukr.anarchist_support@var:v
                    value = 100
                    compare = less_than
                }
            }
            add_to_variable = { ukr.anarchist_support@var:v = ukr.anarchist_support_daily@var:v }
            clamp_variable = {
                var = ukr.anarchist_support@var:v
                min = 0
                max = 100
            }
        }
        if = {
            limit = {
                var:v = {
                    is_core_of = UKR
                }
            }
            if = {
                limit = {
                    var:v = {
                        has_dynamic_modifier = {
                            modifier = ukr_anarchist_state_non_converted
                        }
                    }
                }
                var:v = {
                    remove_dynamic_modifier = {
                        modifier = ukr_anarchist_state_non_converted
                    }
                }
            }
            if = {
                limit = {
                    var:v = {
                        has_dynamic_modifier = {
                            modifier = ukr_anarchist_state_converted
                        }
                    }
                }
                var:v = {
                    remove_dynamic_modifier = {
                        modifier = ukr_anarchist_state_converted
                    }
                }
            }
        }
        else = {
            if = {
                limit = {
                    check_variable = {
                        var = ukr.anarchist_support@var:v
                        value = 100
                        compare = equals
                    }
                    var:v = {
                        NOT = {
                            has_dynamic_modifier = {
                                modifier = ukr_anarchist_state_converted
                            }
                        }
                    }
                }
                var:v = {
                    add_dynamic_modifier = {
                        modifier = ukr_anarchist_state_converted
                    }
                }
                if = {
                    limit = {
                        var:v = {
                            has_dynamic_modifier = {
                                modifier = ukr_anarchist_state_non_converted
                            }
                        }
                    }
                    var:v = {
                        remove_dynamic_modifier = {
                            modifier = ukr_anarchist_state_non_converted
                        }
                    }
                }
            }
            else_if = {
                limit = {
                    check_variable = {
                        var = ukr.anarchist_support@var:v
                        value = 100
                        compare = less_than
                    }
                    var:v = {
                        NOT = {
                            has_dynamic_modifier = {
                                modifier = ukr_anarchist_state_non_converted
                            }
                        }
                    }
                }
                var:v = {
                    add_dynamic_modifier = {
                        modifier = ukr_anarchist_state_non_converted
                    }
                }
                if = {
                    limit = {
                        var:v = {
                            has_dynamic_modifier = {
                                modifier = ukr_anarchist_state_converted
                            }
                        }
                    }
                    var:v = {
                        remove_dynamic_modifier = {
                            modifier = ukr_anarchist_state_converted
                        }
                    }
                }
            }
        }

        if = {
            limit = {
                NOT = {
                    check_variable = {
                        var = ukr.anarchist_support_boost@var:v
                        value = 0
                        compare = equals
                    }
                }
            }
            subtract_from_variable = { ukr.anarchist_support_boost@var:v = 1 }
            clamp_variable = {
                var = ukr.anarchist_support_boost@var:v
                min = 0
                max = 30
            }
        }
    }

    set_temp_variable = { ukr_daily_anarchy_gain_temp = 0 }

    for_each_loop = {
        array = ukr.anarchist_states
        value = v
        index = i
        break = break_array


        if = {
            limit = {
                check_variable = {
                    var = ukr.anarchist_support@var:v
                    value = 100
                    compare = equals
                }
                var:v = {
                    is_core_of = UKR
                }
            }
            add_to_temp_variable = { ukr_daily_anarchy_gain_temp = 0.05 }
        }
    }

    set_variable = { ukr_anarchist_daily_gain = var:ukr_daily_anarchy_gain_temp }
    force_update_dynamic_modifier = yes
}

#Update the state cost

        value = v
        index = i

                var = state_boost_cost
                min = 25

            }

                var = state_core_cost

            }

        }
    }
}
