UKR_dont_push_too_hard_scw  = {
	allowed = {
		original_tag = UKR
	}
	
	enable = {
		ukr_in_scw = yes
		has_active_mission = UKR_anarchism_timeout
		num_of_controlled_states > 19
		check_variable = {
			var = ukr.num_core_states
			value = 20
			compare = less_than
		}
		any_enemy_country = {
			strength_ratio = {
				tag = UKR
				ratio > 1
			}
		}
	}
	
	abort_when_not_enabled = yes
	
	ai_strategy = {
		type = front_control
		
		priority = 1000

		state_trigger = {
			OR = {
				state = 195 #Hold onto Stalingrad
				any_neighbor_state = {
					is_controlled_by = UKR
				}
			}
		}

		execute_order = no
		manual_attack = yes
	}
	
	ai_strategy = {
		type = front_unit_request
	
		state_trigger = {
			OR = {
				state = 195 #Hold onto Stalingrad
				any_neighbor_state = {
					is_controlled_by = UKR
				}
			}
		}
		
		value = 1000
	}
}
