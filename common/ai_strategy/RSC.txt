
TAN_recover_core_territory = {

	reversed = yes

	target_array = neighbors

	enable = {
		NOT = {
            is_in_faction_with = TAN
            is_subject_of = TAN
        }
		strength_ratio = {
			tag = TAN
			ratio < 1.25
		}
        if = {
			limit = {
				is_subject = yes
			}
			OVERLORD = {
				strength_ratio = {
					tag = TAN
					ratio < 1.25
				}
			}
		}
		OR = {
            any_owned_state = {
				is_core_of = TAN
			}
			any_controlled_state = {
				is_core_of = TAN
			}
        }
	}

	enable_reverse = {
        NOT = {
            has_global_flag = second_russian_civil_war_in_progress
        }
		TAN = {
			has_war = no
			is_subject = no
		}
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = conquer
		id = TAN
		value = 9999
	}

    ai_strategy = {
		type = consider_weak
