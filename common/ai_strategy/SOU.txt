SOU_chill_scw  = {

	allowed = {
		original_tag = SOU
	}

	enable = {
        has_global_flag = second_russian_civil_war_in_progress
		date < 1940.3.1 #Just attack at this point
		sov_in_scw = yes
		sou_in_scw = yes
		set_temp_variable = { sov_gun_deficit = var:sov.num_target_equipment_in_armies_k@infantry_equipment }
		subtract_from_temp_variable = { sov_gun_deficit = var:sov.num_equipment_in_armies_k@infantry_equipment }
		set_temp_variable = { sou_gun_deficit = var:sou.num_target_equipment_in_armies_k@infantry_equipment }
		subtract_from_temp_variable = { sou_gun_deficit = var:sou.num_equipment_in_armies_k@infantry_equipment }
		check_variable = {
			var = sou_gun_deficit
			value = sov_gun_deficit
			compare = greater_than
		}
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = front_control

		priority = 5000

		state_trigger = {
			SOV = {
				PREV = {
					is_in_home_area = yes
				}
			}
		}

		tag = SOV

		execute_order = no
		manual_attack = yes
	}
}
