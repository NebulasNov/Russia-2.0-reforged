	on_war_relation_added = {
		effect = {
			if = {
				limit = {

						ROOT = {

								limit = {
									tag = UKR
								}

							}

						FROM = {
							OR = {

							if = {
								limit = {
									tag = UKR
								}

							}
						}
					}
				}
				if = {
					limit = {
						ROOT = {
							has_war_with = FROM
						}
						NOT = {
							is_in_array = { global.russian_scw_participants = ROOT.id }
						}
					}
					add_to_array = { global.russian_scw_participants = ROOT.id }
					ROOT = {
						if = {
							limit = {
								OR = {
									tag = SOV
									is_in_faction_with = SOV
								}
							}
							add_to_array = { global.russian_scw_participants_sov = ROOT.id }
						}
						else_if = {
							limit = {
								OR = {
									tag = SOU
									is_in_faction_with = SOU
								}
							}
							add_to_array = { global.russian_scw_participants_sou = ROOT.id }
						}
						else_if = {
							limit = {
								OR = {
									tag = UKR
									is_in_faction_with = UKR
								}
							}
							add_to_array = { global.russian_scw_participants_ukr = ROOT.id }
						}
						else = {
							add_to_array = { global.russian_scw_participants_other = ROOT.id }
						}
					}
				}
				if = {
					limit = {
						FROM = {
							has_war_with = ROOT
						}
						NOT = {
							is_in_array = { global.russian_scw_participants = FROM.id }
