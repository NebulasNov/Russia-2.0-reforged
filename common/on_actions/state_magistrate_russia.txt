on_actions = {

	#SOV puppet State Magistrate
	on_puppet = {
		effect = {
			if = {
				limit = {
					FROM = {
						tag = SOV
						has_idea = SOV_Impose_order
					}
				}
				ROOT = {
					create_country_leader = {
						name = "State Magistrate"
						desc = "POLITICS_STATE_MAGISTRATE_DESC"
						picture = "gfx/leaders/EUR/Provisional_Government_Fascism.dds"
						expire = "1965.1.1"
						ideology = fascism_ideology
						traits = {
							
						}
					}
				}
			}
		}
	}

	#Create Soviet Republic if Release Country 
	on_release_as_puppet = {
		effect = {
			if = {
				limit = {
					FROM = {
						tag = SOV
						has_completed_focus = SOV_great_purge2
					}
				}
				FROM = {
					set_autonomy = {
						target = ROOT
						autonomy_state = autonomy_soviet_republic
					}
				}
			}
		}
	}
}
