on_actions = {

	#SOU Gets Annexed By SOV and Receives Events
	on_capitulation = {
		effect = {
			if = {
				limit = {
					ROOT = {
						tag = SOU
					}
					FROM = {
						tag = SOV
						has_country_flag = black_revolution_civil_war_in_progress
					}
				}
				ROOT = {
					transfer_navy = {
						target = FROM
					}
				}	
				every_state = {
					limit = {
						is_owned_by = ROOT
					}
					FROM = {
						transfer_state = PREV
					}
				}
				FROM = {
					country_event = { id = soviets.412 hours = 6 }
					#news_event = { id = newsrussia2.67 days = 1 }
				}
			}
		}
	}
