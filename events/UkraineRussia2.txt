			}
		}

			target = SOU
			type = annex_everything
		}
		hidden_effect = {
			SOU = {
				country_event = { id = ukrains.38 days = 1 }

							}
						}
					}
					ROOT = {
						declare_war_on = {
							target = PREV
							type = second_russian_civil_war
						}
						add_ai_strategy = {
							type = conquer
							id = PREV
							value = 200
						}
					}
				}
			}

			country_event = ukrains.50
		}
