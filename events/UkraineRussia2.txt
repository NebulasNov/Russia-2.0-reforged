	id = russia2_ukrain.10
	title = russia2_ukrain.10.t
	desc = russia2_ukrain.10.d
	picture = GFX_battkle_kursk_russia2

	fire_only_once = yes

	trigger = {
		tag = UKR
		has_war = no
		is_subject = no
		has_global_flag = second_russian_civil_war_in_progress
		NOT = {
			has_government = anarchy
			has_country_flag = ukr_anarchist_rebels
		}
		SOU = {
			has_war_with = SOV
			surrender_progress > 0.65
		}
	}

	mean_time_to_happen = { days = 1 }

	option = { #Attack SOU
		name = russia2_ukrain.10.a
		ai_chance = { factor = 25 }
		declare_war_on = {
			target = SOU
			type = annex_everything
		}
		hidden_effect = {
			SOU = {
				country_event = { id = russia2_ukrain.38 days = 1 }
			}
			for_each_scope_loop = {
				array = global.russian_scw_participants_sou

				if = {
					limit = {
						NOT = {
							ROOT = {
								has_war_with = PREV
							}
						}
					}
					ROOT = {
						declare_war_on = {
							target = PREV
							type = second_russian_civil_war
						}
						add_ai_strategy = {
							type = conquer
							id = PREV
							value = 200
						}
					}
				}
			}
		}
		add_ideas = SOV_Second_civil_war
		if = {
			limit = {
				has_country_flag = has_accepted_scw_neutrality
			}
			custom_effect_tooltip = sov_might_take_action_sou
			hidden_effect = {
				SOV = {
					country_event = { id = russia2_ukrain.11 hours = 1 }
				}
			}
		}
	}

	option = { #Alliance with Finland
		name = russia2_ukrain.10.b
		trigger = {
			FIN = {
				exists = yes
				NOT = {
					has_war = yes
					is_faction_leader = yes
					is_subject = yes
				}
			}
			any_of_scopes = {
				array = global.russian_scw_participants

				is_neighbor_of = FIN
			}
		}
		ai_chance = { factor = 25 }
		declare_war_on = {
			target = SOU
			type = annex_everything
		}
		hidden_effect = {
			SOU = {
				country_event = { id = russia2_ukrain.38 days = 1 }
			}
			for_each_scope_loop = {
				array = global.russian_scw_participants_sou

				if = {
					limit = {
						NOT = {
							ROOT = {
								has_war_with = PREV
							}
						}
					}
					ROOT = {
						declare_war_on = {
							target = PREV
							type = second_russian_civil_war
						}
						add_ai_strategy = {
							type = conquer
							id = PREV
							value = 200
						}
					}
				}
			}
		}
		add_ideas = SOV_Second_civil_war
		FIN = {
			country_event = russia2_ukrain.50
		}
		custom_effect_tooltip = ukr_will_make_concessions_fin
		if = {
			limit = {
				has_country_flag = has_accepted_scw_neutrality
			}
			custom_effect_tooltip = sov_might_take_action_sou
			hidden_effect = {
				SOV = {
					country_event = { id = russia2_ukrain.11 hours = 1 }
				}
			}
		}
	}

	option = { #Do nothing
		name = russia2_ukrain.10.c
		ai_chance = { factor = 50 }
		add_stability = 0.05
	}
}
