#Country Gets Event to Give Back Bulgarian Claim
country_event = {
	id = bulgarias.4
	title = bulgarias.4.t
	desc = bulgarias.4.d
	picture = GFX_report_event_stalin_02
	
	is_triggered_only = yes

    immediate = {
        set_country_flag = receive_event_bulgaria
    }
	
	option = { #Okay
		ai_chance = {
			factor = 100
            modifier = {
                factor = 45
                OR = {
                    tag = YUG
                    tag = GRE
                    tag = MAC
                }
            }
		}
		name = bulgarias.4.a
        every_state = {
            limit = {
                is_controlled_by = ROOT
                has_state_flag = state_claimed_bulgaria_slavic_alliance
            }
            clr_state_flag = state_claimed_bulgaria_slavic_alliance
            BUL = {
                transfer_state = PREV
            }
        }
        if = {
            limit = {
                all_owned_state = {
                    has_state_flag = state_claimed_bulgaria_slavic_alliance
                }
            }
            custom_effect_tooltip = SOV_game_totally_over_TT
        }
        BUL = {
            country_event = { id = bulgarias.5 days = 1 }
        }
        clr_country_flag = receive_event_bulgaria
	}

    option = { #No
		ai_chance = {
			factor = 0
            modifier = {
                factor = 55
                OR = {
                    tag = YUG
                    tag = GRE
                    tag = MAC
                }
            }
		}	
		name = bulgarias.4.b
        BUL = {
            country_event = { id = bulgarias.6 days = 1 }
        }
        custom_effect_tooltip = BUL_leave_faction_refuse_claim_tt
	}
}

country_event = {
	id = bulgarias.6
	title = bulgarias.6.t
	desc = bulgarias.6.d
	picture = GFX_report_event_stalin_02
	
	is_triggered_only = yes
	
	option = { #Okay, give up our claim...
		ai_chance = {
			factor = 20
		}	
		name = bulgarias.6.a
        every_state = {
            limit = {
                is_controlled_by = FROM
                has_state_flag = state_claimed_bulgaria_slavic_alliance
            }
            clr_state_flag = state_claimed_bulgaria_slavic_alliance
            BUL = {
                remove_state_claim = PREV
            }
        }
        FROM = {
            country_event = { id = bulgarias.8 days = 1 }
        }
	}

    option = { #Leave the faction
		ai_chance = {
			factor = 80
		}
		name = bulgarias.6.b
        leave_faction = yes
        set_country_flag = SOV_BUL_betrayed_claims
        SOV = {
            country_event = { id = bulgarias.7 days = 1 }
        }
        BUL = {
            add_ai_strategy = {
                type = antagonize
                id = FROM
                value = 200
            } 
        }
        FROM = {
            add_ai_strategy = {
                type = antagonize
                id = BUL
                value = 200
            }
        }
	}
}
